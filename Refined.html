// --- NAVIGATION HISTORY LOGIC ---
let historyStack = ['home'];
function navigateTo(pageId, isBack = false) {
document.querySelectorAll('.page').forEach(page =>
page.classList.remove('active'));
document.getElementById(pageId).classList.add('active');
window.scrollTo({ top: 0, behavior: 'smooth' });
if (!isBack && historyStack[historyStack.length - 1] !==
pageId) {
historyStack.push(pageId);
}
}
function goBack() {
if (historyStack.length > 1) {
historyStack.pop(); // Remove current
const prevPage = historyStack[historyStack.length -
1];
navigateTo(prevPage, true);
} else {
navigateTo('home');
}
}

function validateRegPassword() {
const pass = document.getElementById('reg-password');
const val = pass.value;
const hasUpper = /[A-Z]/.test(val);
const hasLower = /[a-z]/.test(val);
const hasNum = /[0-9]/.test(val);
const hasSpec =
/[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]+/.test(val);
const isLong = val.length >= 6;
const isValid = hasUpper && hasLower && hasNum && hasSpec
&& isLong;
if (val.length === 0) {
pass.classList.remove('valid-field', 'invalid-field');
return false;
}
if (isValid) {
pass.classList.add('valid-field');
pass.classList.remove('invalid-field');
return true;
} else {
pass.classList.add('invalid-field');
pass.classList.remove('valid-field');
return false;
}
}
function generatePassword() {
const name = document.getElementById('reg-name').value;
const email = document.getElementById('reg-email').value;
if (!name || !email) {
alert("Please fill out Name and Email before
generating a password.");
return;
}
const chars = "abcdefghijklmnopqrstuvwxyz";
const upper = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
const nums = "0123456789";
const spec = "!@#$%^&*";
let pass = "";
pass += upper[Math.floor(Math.random() * upper.length)];
pass += spec[Math.floor(Math.random() * spec.length)];
pass += nums[Math.floor(Math.random() * nums.length)];
for(let i = 0; i < 7; i++) {
const pool = chars + upper + nums + spec;
pass += pool[Math.floor(Math.random() * pool.length)];
}
// Shuffle
pass = pass.split('').sort(() => 0.5 -
Math.random()).join('');
const passInput = document.getElementById('reg-password');
passInput.value = pass;
// Switch to text type momentarily to show it
passInput.type = "text";
document.querySelector('#register .eye-icon').innerText =
"
🙈
";
validateRegPassword();
}
function handleRegister(e) {

if (!validateRegPassword()) {
alert("Password does not meet the strict criteria
(Upper, Lower, Number, Special, Min 6).");
return;
}
alert("Account details saved to database successfully!");
navigateTo('home');
}
